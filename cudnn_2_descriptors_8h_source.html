16 /*
17 Note [cuDNN dropout descriptor initialization]
18 ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
19 
20 In most cases, setting descriptors in cuDNN is cheap (e.g.,
21 cudnnSetTensorNdDescriptor). However, this is not the case for
22 cudnnSetDropoutDescriptor: in cuDNN 6/7 (and possibly others) it does an
23 expensive precomputation to initialize the random number generator states. In
24 cuDNN 6, this is the ONLY official mechanism to initialize a dropout descriptor,
25 which means that law-abiding clients were expected to generate a dropout
26 descriptor once and cache it. However, our ATen interface is (1) stateless (so
27 we can't cache the descriptors) and (2) does not accept arbitrary user types in
28 its interface (so we can't pass the descriptor in). This puts us in a pickle.
29 
30 In cuDNN 7, a new function, cudnnRestoreDropoutDescriptor was added, which
31 forgoes the expensive initialization process, and can initialize the
32 descriptor with a pre-initialized state CUDA tensor. This is great, because
33 it means we can simply pass in the state tensor and then initialize the
34 descriptor internally. Unfortunately, this function is not available in
35 cuDNN 6.
36 
37 To work around this, we break the cuDNN abstraction barrier, and have
38 the struct layout of the underlaying dropout descriptor. With this struct,
39 we can reimplement cudnnRestoreDropoutDescriptor from scratch. Great!
40 */

64 // The stride for a size-1 dimensions is not uniquely determined; in
65 // fact, it can be anything you want, because the fact that the
66 // tensor is size 1 at this dimension means that you will never actually
67 // try advancing your pointer by this stride.
68 //
69 // However, CuDNN has a much more stringent requirement on strides:
70 // if you are passing a contiguous input, it better be the case
71 // that the stride for dim i is the product of the sizes of dims
72 // i+1 to the end. This stride is indeed uniquely determined. This
73 // function modifies 'stride' in place so this invariant holds.
74 static inline void fixSizeOneDimStride(int dim, const int *size, int *stride) {
75  int64_t z = 1;
76  for(int d = dim-1; d >= 0; d--)
77  {
78  if (size[d] == 1) {
79  stride[d] = z;
80  } else {
81  z *= size[d];
82  }
83  }
84 }

95 // A generic class for wrapping cuDNN descriptor types. All you need
96 // is to give the underlying type the Descriptor_t points to (usually,
97 // if it's cudnnTensorDescriptor_t it points to cudnnTensorStruct),
98 // the constructor and the destructor. Subclasses are responsible
99 // for defining a set() function to actually set the descriptor.
100 //
101 // Descriptors default construct to a nullptr, and have a descriptor
102 // initialized the first time you call set() or any other initializing
103 // function.
104 template <typename T, cudnnStatus_t (*ctor)(T**), cudnnStatus_t (*dtor)(T*)>
105 class AT_CUDA_API Descriptor
106 {
107 public:
108  // TODO: Figure out why const-correctness doesn't work here
109 
110  // Use desc() to access the underlying descriptor pointer in
111  // a read-only fashion. Most client code should use this.
112  // If the descriptor was never initialized, this will return
113  // nullptr.
114  T* desc() const { return desc_.get(); }
115  T* desc() { return desc_.get(); }
116 
117  // Use mut_desc() to access the underlying desciptor pointer
118  // if you intend to modify what it points to (e.g., using
119  // cudnnSetFooDescriptor). This will ensure that the descriptor
120  // is initialized. Code in this file will use this function.
121  T* mut_desc() { init(); return desc_.get(); }
122 protected:
123  void init() {
124  if (desc_ == nullptr) {
125  T* raw_desc;
126  AT_CUDNN_CHECK(ctor(&raw_desc));
127  desc_.reset(raw_desc);
128  }
129  }
130 private:
131  std::unique_ptr<T, DescriptorDeleter<T, dtor>> desc_;
132 };

218 inline cudnnStatus_t cudnnRestoreDropoutDescriptor(
219  cudnnDropoutDescriptor_t dropoutDesc,
220  cudnnHandle_t handle,
221  float dropout,
222  void *states,
223  size_t stateSizeInBytes,
224  unsigned long long seed) {
225  // Try to accurately simulate cuDNN's behavior, for our cuDNN 6 friends.
226  // This is not entirely accurate but is good enough to catch some API
227  // uses which would not be compatible in cuDNN 7. Feel free to fix
228  // this if you notice something is wrong.
229  if (states == nullptr) return CUDNN_STATUS_INVALID_VALUE;
230  if (stateSizeInBytes == 0) return CUDNN_STATUS_INVALID_VALUE;
231  size_t expectedStateSizeInBytes;
232  // State size will differ depending on size of GPU
233  auto ret = cudnnDropoutGetStatesSize(handle, &expectedStateSizeInBytes);
234  if (ret != CUDNN_STATUS_SUCCESS) return ret;
235  if (expectedStateSizeInBytes != stateSizeInBytes) return CUDNN_STATUS_INVALID_VALUE;
236  dropoutDesc->dropout = dropout;
237  dropoutDesc->nstates = (int)stateSizeInBytes/sizeof(curandState_t);
238  dropoutDesc->states = states;
239  return CUDNN_STATUS_SUCCESS;
240 }

255  AT_ASSERTM(dropout > 0, "dropout must be nonzero; otherwise call set_no_dropout");

262  }
