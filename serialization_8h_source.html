200 void save(Archive& archive, torch::Tensor const& tensor) {
201  torch::NoGradGuard guard;
202  if (!tensor.defined()) {
203  int32_t typeId = ::torch::detail::scalarTypeId(torch::Dtype::Undefined);
204  archive(CEREAL_NVP(typeId));
205  return;
206  } else {
207  int32_t typeId = ::torch::detail::scalarTypeId(tensor.dtype());
208  archive(CEREAL_NVP(typeId));
209  }
210  auto sizes = std::vector<int64_t>();
211  auto buf = std::vector<uint8_t>();
212  for (auto s : tensor.sizes()) {
213  sizes.push_back(s);
214  }
215  auto contig = tensor.toBackend(torch::kCPU).contiguous();
216  int32_t backend = ::torch::detail::backendId(tensor.data().type().backend());
217 
218  archive(CEREAL_NVP(backend), CEREAL_NVP(sizes));
219  agimpl::saveBinary(
220  archive,
221  contig.data_ptr(),
222  tensor.numel() * tensor.data().type().elementSizeInBytes());
223 }

233  torch::NoGradGuard guard;
234  torch::Dtype type;
235  int32_t typeId;
236  archive(CEREAL_NVP(typeId));
237  type = ::torch::detail::scalarTypeFromId(typeId);
238  if (type == torch::Dtype::Undefined) {
239  tensor = torch::Tensor();
240  return;
241  }
242 
243  int32_t backendId;
244  auto sizes = std::vector<int64_t>();
245  auto buf = std::vector<uint8_t>();
246  archive(CEREAL_NVP(backendId), CEREAL_NVP(sizes));
247 
248  at::Backend backend = ::torch::detail::backendFromId(backendId);
249  if (!tensor.defined() || tensor.dtype() != type) {
250  tensor = torch::empty({}, torch::getType(backend, type));
251  }
252  tensor.data().resize_(sizes);
253
